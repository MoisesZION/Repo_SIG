Funcion suma<-SumarNPares(N)
	Definir suma,inicio,i Como Entero
	Si N%2=0 Entonces
		inicio<-N+2
	SiNo
		inicio<-N+1
	FinSi
	suma<-0
	Para i<-1 Hasta N Hacer
		suma<-suma+inicio
		inicio<-inicio+2
	FinPara
FinFuncion
